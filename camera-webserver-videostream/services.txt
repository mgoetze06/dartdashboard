[Unit]
Description=Shutting Down the Device when battery is discharging
After=network.target

[Service]
WorkingDirectory=/home/boris/webserver
ExecStart=/usr/bin/python3 /home/boris/webserver/poweroff.py

[Install]
WantedBy=multi-user.target





######################################################################################
Service erstellen:
sudo systemctl --force --full edit boris.service
